        //changed the increment to 2 in this for loop because the spaces are part of the grid//

    function countBs(word){
    //empty string that will hold the answers
    var emptyString="";
    //for loop that starts at zero or the first letter in the string and stops when it is longer than the word//
    for(var count=0; count<=word.length; count++){
      var answer2=word.charAt(count);
    //if statement that will skip to next letter in loop if it isnt equal to B or will add it to the empty variable q if it is//
      if(answer2!=="B"){
      continue;
      }
        else{
    //This will write the length of the empty variable emptyString out to the target element #answer4//
        emptyString+=answer2;
        $("#answer4").html("The number of B's in this word: "+ emptyString.length);
        }
    }
    };

            $('#question4').on("click", function(){
        countBs("BBC");
    });
